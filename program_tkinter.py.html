import tkinter as tk
from tkinter import messagebox

users = []

def add_user():
    name = entry.get()
    if name:
        users.append(name)
        update_list()
        entry.delete(0, tk.END)
        messagebox.showinfo("Éxito",f"Usuario {name} añadido")
    else:
        messagebox.showwarning("Atención","Debes escribir un nombre")

def delete_user():
    name = entry.get()
    if name in users:
        users.remove(name)
        update_list()
        entry.delete(0, tk.END)
        messagebox.showinfo("Éxito",f"Usuario {name} eliminado")
    else:
        messagebox.showerror("Error","Ese usuario no existe")

def update_list():
    listbox.delete(0, tk.END)
    for user in users:
        listbox.insert(tk.END, user)

def exit_app():
    root.destroy()

root = tk.Tk()
root.title("Gestor de Usuarios")
root.geometry("300x300")

entry = tk.Entry(root)
entry.pack(pady=5)

btn_add = tk.Button(root, text="Añadir usuario",command=add_user)
btn_add.pack(pady=5)

btn_delete = tk.Button(root, text="Eliminar usuario", command=delete_user)
btn_delete.pack(pady=5)

btn_exit = tk.Button(root, text="Salir", command=exit_app)
btn_exit.pack(pady=5)

listbox = tk.Listbox(root)
listbox.pack(pady=10, fill=tk.BOTH,expand=True)

root.mainloop()
